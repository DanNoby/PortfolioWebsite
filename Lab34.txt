#include <stdio.h>
#include <string.h>
#define STATES 20
#define IP 20
#define MAX (STATES + 1)

int nfa[STATES][IP][MAX];
int visited[STATES]; 
int num_states, num_cols;

void find_closure(int s) {
    visited[s] = 1;
    for (int i = 0; nfa[s][0][i] != -1; i++) {
        int next = nfa[s][0][i];
        if (next >= 0 && next < STATES && !visited[next]) {
            find_closure(next);
        }
    }
}

int main() {
    memset(nfa, -1, sizeof(nfa));

    printf("Enter number of States: ");
    scanf("%d", &num_states);
    
    printf("Enter number of Input Columns (including Epsilon): ");
    scanf("%d", &num_cols);

    // Input Loop
    for (int i = 0; i < num_states; i++) {
        for (int j = 0; j < num_cols; j++) {
            printf("State %d, IP %d (end with -1): ", i, j);
            
            int k = 0, val;
            while (1) {
                scanf("%d", &val);
                nfa[i][j][k++] = val;
                if (val == -1) break;
            }
        }
    }

    for (int i = 0; i < num_states; i++) {
        memset(visited, 0, sizeof(visited));
        find_closure(i);

        printf("State %d: { ", i);
        for (int j = 0; j < STATES; j++) {
            if (visited[j]) printf("%d ", j);
        }
        printf("}\n");
    }
    return 0;
}
